Inputs

accordionStyle
Type : AccordionStyle
Default value : { background: false, border: false }

Style to be defined for accordion header to show/ hide background and border. if background and border as true, shows the default background color and border will be applied. if background and border as false, shows the default background color as white and border will not be applied.
container
Type : string

Id of the accordion
disableAccordion
Type : boolean
Default value : false

boolean to expand/collapse accordion
groups
Type : Array<AttributeGroup | OptionGroup>

Array of attribute groups or option groups to be rendered in accordion
items
Type : ConfigurationResponse
parentId
Type : string

parent Id to render child accordions

 
<div *ngIf="groups?.length > 0">
    <ng-template [ngTemplateOutlet]="accordionHeader"
        [ngTemplateOutletContext]="{groups: groups, container: container, accordionStyle: accordionStyle, items: items, parentId: parentId}">
    </ng-template>
</div>

<ng-template #accordionHeader let-groups="groups" let-container="container" let-accordionStyle="accordionStyle"
    let-items="items" let-parentId="parentId">
    <ng-container *ngFor="let group of groups">
        <div *ngIf="!parentId" class="card" [class.border-0]="!accordionStyle?.border" [hidden]="group.hide">
            <!--*ngIf="!group.hidden"  -->
            <div class="card-header px-2 py-1" [class.bg-light]="!accordionStyle?.background"
                [class.border-1]="!accordionStyle?.border">
                <h5 class="mb-0">
                    <button class="btn btn-link chevron" type="button" data-toggle="collapse"
                        [attr.data-target]="'#a'+ group?.groupExtId"
                        [attr.aria-expanded]="disableAccordion ? 'true' : 'false'">
                        {{group?.name}}
                    </button>
                </h5>
            </div>
            <div [id]="'a'+ group?.groupExtId" class="collapse pl-4"
                [attr.data-parent]="(!disableAccordion) ? '#'+container  : null"
                [class.show]="disableAccordion">
                <ng-template [ngTemplateOutlet]="accordionBody"
                    [ngTemplateOutletContext]="{group: group, container: container, disableAccordion: disableAccordion, items: items}">
                </ng-template>
            </div>
        </div>
        <div *ngIf="parentId" class="card" [class.border-0]="!accordionStyle?.border" [hidden]="group.hide">
            <!--*ngIf="!group.hidden"  -->
            <div class="card-header px-2 py-1" [class.bg-light]="!accordionStyle?.background"
                [class.border-1]="!accordionStyle?.border">
                <h5 class="mb-0">
                    <button class="btn btn-link chevron" type="button" data-toggle="collapse"
                        [attr.data-target]="'#a'+parentId +'-'+ group?.groupExtId" [attr.aria-expanded]="false">
                        {{group?.name}}
                    </button>
                </h5>
            </div>
            <div [id]="'a' + parentId +'-'+ group?.groupExtId" class="collapse pl-4" [attr.data-parent]="'#' + parentId">
                <ng-template [ngTemplateOutlet]="accordionBody"
                    [ngTemplateOutletContext]="{group: group, container: container, disableAccordion: disableAccordion, items: items, parentId: parentId}">
                </ng-template>
            </div>
        </div>
    </ng-container>
</ng-template>

<ng-template #accordionBody let-group="group" let-container="container" let-disableAccordion="disableAccordion"
    let-items="items" let-parentId="parentId">
    <div class="card-body">
        <!-- load the attributes -->
        <apt-cms-product-attribute *ngIf="group?.attributes" [attributes]="group?.attributes"
            [attributeValue]="group?.productAttributeValue" [items]="items"
            [numberOfAttributeColumns]="group?.attributeColumns">
        </apt-cms-product-attribute>

        <!-- load the options -->
        <apt-cms-product-option [options]="group?.options" [container]="parentId" [items]="items"
            [disableAccordion]="disableAccordion">
        </apt-cms-product-option>

        <!-- load the child option groups -->
        <div  class="accordion" [id]="parentId ? parentId +'-'+ group?.groupExtId : group?.groupExtId">
            <ng-template *ngIf="group?.children?.length > 0" [ngTemplateOutlet]="accordionHeader"
                [ngTemplateOutletContext]="{groups: group?.children, accordionStyle: {background:false, border:false}, items: items, parentId: parentId ? parentId +'-'+ group?.groupExtId : group?.groupExtId}">
            </ng-template>
        </div>
    </div>
</ng-template>
